关于原型链的理解

题

    function A(){
      this.x=10
      this.y=20
      this.f=function(){
          console.log(this.x+=20);
          this.__proto__.f()
        }
    }
    A.prototype.x=1
    A.prototype.y=2
    A.prototype.f=function(){
        console.log(this.y+=30)
    }
    function B(){
      this.x=11
      this.y=22
      this.g=function(){
          console.log(this.x+=this.y+=100);
          this.f();
          this.__proto__.f()
        }
    }
    B.prototype=new A()
    B.prototype.h=function(){
        this.g();
        this.f();
        this.__proto__.f()
    }
    var b=new B
    b.h();b.h()
