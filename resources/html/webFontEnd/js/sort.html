冒泡排序

两项之间比较，较小值向前交换位置，注意比较边界

                function bubbleSort(data){
                    var minPos = 0;
                    for(var i=0;i<data.length;i++){
                        for(var j=0;j<data.length-i-1;j++){
                            if(data[j+1]<data[j]){
                                minPos = data[j+1];
                                data[j+1] = data[j];
                                data[j] = minPos;
                            }
                        }
                    }
                }
插入排序

将项插入到一个有序的队列中去，仍保证队列顺序无误,从队列尾部递减比较

                function insertSort(data){
                    var minPos = 0;
                    for(var i=0;i<data.length;i++){
                        for(var j=i;j>=0;j--){
                            if(data[i]<data[j]){
                                minPos = data[i];
                                data[i] = data[j];
                                data[j] = minPos;
                            }
                        }
                    }
                }

快速排序

寻找一个参考值，数组左边小于参考值，右边大于参考值，建立边界，向内递归

                function quickSort(data){
                    var low = 0;
                    var high = data.length-1;
                    var sort = function(low,high){
                        if(low>high){
                            return;
                        }
                        var tmpLow = low;
                        var tmpHigh = high;
                        var key = data[high];
                        while(low<high){
                            while(low<high&&data[low]<key){
                                low++
                            }
                            data[high] = data[low];
                            while(low<high&&key<data[high]){
                                high--;
                            }
                            data[low] = data[high];
                        }
                        data[low] = key;
                        sort(tmpLow,low-1);
                        sort(high+1,tmpHigh);
                    }
                    sort(low,high);
                }
